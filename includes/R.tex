%
%	LATIN LETTER SMALL CAPITAL R							(U+0280)
	\pgfparserdef{tcparser}{ini}r {\TC@write{ʀ}}
	\pgfparserdef{tcparser}{ini}R {\TC@write{R}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING ACUTE ACCENT						(U+0301)
		\pgfparserdef{tcparser}{ini}ŕ {\TC@write{ʀ́}}
		\pgfparserdef{tcparser}{ini}Ŕ {\TC@write{Ŕ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING DOT ABOVE							(U+0307)
		\pgfparserdef{tcparser}{ini}ṙ {\TC@write{ʀ̇}}
		\pgfparserdef{tcparser}{ini}Ṙ {\TC@write{Ṙ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING CARON								(U+030C)
		\pgfparserdef{tcparser}{ini}ř {\TC@write{ʀ̌}}
		\pgfparserdef{tcparser}{ini}Ř {\TC@write{Ř}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING DOUBLE GRAVE ACCENT				(U+030F)
		\pgfparserdef{tcparser}{ini}ȑ {\TC@write{ʀ̏}}
		\pgfparserdef{tcparser}{ini}Ȑ {\TC@write{Ȑ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING INVERTED BREVE					(U+0311)
		\pgfparserdef{tcparser}{ini}ȓ {\TC@write{ʀ̑}}
		\pgfparserdef{tcparser}{ini}Ȓ {\TC@write{Ȓ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING DOT BELOW							(U+0323)
		\pgfparserdef{tcparser}{ini}ṛ {\TC@write{ʀ̣}}
		\pgfparserdef{tcparser}{ini}Ṛ {\TC@write{Ṛ}}

		%	LATIN LETTER SMALL CAPITAL R					(U+0280)
		%	with	COMBINING DOT BELOW						(U+0323)
		%	and COMBINING MACRON							(U+0304)
			\pgfparserdef{tcparser}{ini}ṝ {\TC@write{ʀ̣̄}}
			\pgfparserdef{tcparser}{ini}Ṝ {\TC@write{Ṝ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING CEDILLA							(U+0327)
		\pgfparserdef{tcparser}{ini}ŗ {\TC@write{ʀ̧}}
		\pgfparserdef{tcparser}{ini}Ŗ {\TC@write{Ŗ}}

	%	LATIN LETTER SMALL CAPITAL R						(U+0280)
	%	with	COMBINING MACRON BELOW						(U+0331)
		\pgfparserdef{tcparser}{ini}ṟ {\TC@write{ʀ̱}}
		\pgfparserdef{tcparser}{ini}Ṟ {\TC@write{Ṟ}}
