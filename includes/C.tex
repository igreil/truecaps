	\pgfparserdef{tcparser}{ini}c {\TC@write{ᴄ}}
	\pgfparserdef{tcparser}{ini}C {\TC@write{C}}

	%	LATIN LETTER SMALL CAPITAL C						(U+1D04)

		\pgfparserdef{tcparser}{ini}Ć {\TC@write{Ć}}

	%	LATIN LETTER SMALL CAPITAL C						(U+1D04)
	%	with	COMBINING CIRCUMFLEX ACCENT					(U+0302)
		\pgfparserdef{tcparser}{ini}ĉ {\TC@write{ᴄ̂}}
		\pgfparserdef{tcparser}{ini}Ĉ {\TC@write{Ĉ}}

	%	LATIN LETTER SMALL CAPITAL C						(U+1D04)
	%	with	COMBINING DOT ABOVE							(U+0307)
		\pgfparserdef{tcparser}{ini}ċ {\TC@write{ᴄ̇}}
		\pgfparserdef{tcparser}{ini}Ċ {\TC@write{Ċ}}

	%	LATIN LETTER SMALL CAPITAL C						(U+1D04)
	%	with	COMBINING CARON								(U+030C)
		\pgfparserdef{tcparser}{ini}č {\TC@write{ᴄ̌}}
		\pgfparserdef{tcparser}{ini}Č {\TC@write{Č}}

	%	LATIN LETTER SMALL CAPITAL C						(U+1D04)
	%	with	COMBINING CEDILLA							(U+0327)
		\pgfparserdef{tcparser}{ini}ç {\TC@write{ᴄ̧}}
		\pgfparserdef{tcparser}{ini}Ç {\TC@write{Ç}}
