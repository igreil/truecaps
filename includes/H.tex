%
%	LATIN LETTER SMALL CAPITAL H							(U+029C)
	\pgfparserdef{tcparser}{ini}h {\TC@write{ʜ}}
	\pgfparserdef{tcparser}{ini}H {\TC@write{H}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING CIRCUMFLEX						(U+0302)
		\pgfparserdef{tcparser}{ini}ĥ {\TC@write{ʜ̂}}
		\pgfparserdef{tcparser}{ini}Ĥ {\TC@write{Ĥ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING DOT ABOVE							(U+0307)
		\pgfparserdef{tcparser}{ini}ḣ {\TC@write{ʜ̇}}
		\pgfparserdef{tcparser}{ini}Ḣ {\TC@write{Ḣ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING DIAERESIS							(U+0308)
		\pgfparserdef{tcparser}{ini}ḧ {\TC@write{ʜ̈}}
		\pgfparserdef{tcparser}{ini}Ḧ {\TC@write{Ḧ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING CARON								(U+030C)
		\pgfparserdef{tcparser}{ini}ȟ {\TC@write{ʜ̌}}
		\pgfparserdef{tcparser}{ini}Ȟ {\TC@write{Ȟ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING DOT BELOW							(U+0323)
		\pgfparserdef{tcparser}{ini}ḥ {\TC@write{ʜ̣}}
		\pgfparserdef{tcparser}{ini}Ḥ {\TC@write{Ḥ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING CEDILLA							(U+0327)
		\pgfparserdef{tcparser}{ini}ḩ {\TC@write{ʜ̧}}
		\pgfparserdef{tcparser}{ini}Ḩ {\TC@write{Ḩ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING BREVE BELOW						(U+032E)
		\pgfparserdef{tcparser}{ini}ḫ {\TC@write{ʜ̮}}
		\pgfparserdef{tcparser}{ini}Ḫ {\TC@write{Ḫ}}

	%	LATIN LETTER SMALL CAPITAL H						(U+029C)
	%	with	COMBINING MACRON BELOW						(U+0331)
		\pgfparserdef{tcparser}{ini}ẖ {\TC@write{ʜ̱}}
		% no corrseponding uppercase character!

%-----------------------------------------------------------------------------

	\pgfparserdef{tcparser}{ini}ħ {\TC@write{ʜ̵̶}}
	\pgfparserdef{tcparser}{ini}Ħ {\TC@write{Ħ}}
