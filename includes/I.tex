
%-----------------------------------------------------------------------------
%	LATIN LETTER SMALL CAPITAL I							(U+026A)
	\pgfparserdef{tcparser}{ini}i {\TC@write{ɪ}}
	\pgfparserdef{tcparser}{ini}I {\TC@write{I}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING GRAVE ACCENT						(U+0300)
		\pgfparserdef{tcparser}{ini}ì {\TC@write{ɪ̀}}
		\pgfparserdef{tcparser}{ini}Ì {\TC@write{Ì}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING ACUTE ACCENT						(U+0301)
		\pgfparserdef{tcparser}{ini}í {\TC@write{ɪ́}}
		\pgfparserdef{tcparser}{ini}Í {\TC@write{Í}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING CIRCUMFLEX ACCENT					(U+0302)
		\pgfparserdef{tcparser}{ini}î {\TC@write{ɪ̂}}
		\pgfparserdef{tcparser}{ini}Î {\TC@write{Î}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING TILDE								(U+0303)
		\pgfparserdef{tcparser}{ini}ĩ {\TC@write{ɪ̃}}
		\pgfparserdef{tcparser}{ini}Ĩ {\TC@write{Ĩ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING MACRON							(U+0304)
		\pgfparserdef{tcparser}{ini}ī {\TC@write{ɪ̄}}
		\pgfparserdef{tcparser}{ini}Ī {\TC@write{Ī}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING BREVE								(U+0306)
		\pgfparserdef{tcparser}{ini}ĭ {\TC@write{ɪ̆}}
		\pgfparserdef{tcparser}{ini}Ĭ {\TC@write{Ĭ}}

% Since a "normal", dotted i will produce an undotted I in caps or as
% uppercase character, we reverse that behavior for the undotted ı, which
% will produce a dotted uppercase İ.

	%	LATIN SMALL LETTER DOTLESS I						(U+0131)
	%	with	COMBINING DOT ABOVE							(U+0307)
		\pgfparserdef{tcparser}{ini}ı {\TC@write{ɪ̇}}
	%	LATIN CAPITAL LETTER I WITH DOT ABOVE				(U+0130)
		\pgfparserdef{tcparser}{ini}İ {\TC@write{İ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING DIAERESIS							(U+0308)
		\pgfparserdef{tcparser}{ini}ï {\TC@write{ɪ̈}}
		\pgfparserdef{tcparser}{ini}Ï {\TC@write{Ï}}

		%	LATIN LETTER SMALL CAPITAL I					(U+026A)
		%	with	COMBINING DIAERESIS 					(U+0308)
		%	and		COMBINING ACUTE ACCENT					(U+0301)
			\pgfparserdef{tcparser}{ini}ḯ {\TC@write{ɪ̈́}}
			\pgfparserdef{tcparser}{ini}Ḯ {\TC@write{Ḯ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING HOOK ABOVE						(U+0309)
		\pgfparserdef{tcparser}{ini}ỉ {\TC@write{ɪ̉}}
		\pgfparserdef{tcparser}{ini}Ỉ {\TC@write{Ỉ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING CARON								(U+030C)
		\pgfparserdef{tcparser}{ini}ǐ {\TC@write{ɪ̌}}
		\pgfparserdef{tcparser}{ini}Ǐ {\TC@write{Ǐ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING DOUBLE GRAVE ACCENT				(U+030F)
		\pgfparserdef{tcparser}{ini}ȉ {\TC@write{ɪ̏}}
		\pgfparserdef{tcparser}{ini}Ȉ {\TC@write{Ȉ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING INVERTED BREVE					(U+0311)
		\pgfparserdef{tcparser}{ini}ȋ {\TC@write{ɪ̑}}
		\pgfparserdef{tcparser}{ini}Ȋ {\TC@write{Ȋ}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING DOT BELOW							(U+0323)
		\pgfparserdef{tcparser}{ini}ị {\TC@write{ɪ̣}}
		\pgfparserdef{tcparser}{ini}Ị {\TC@write{Ị}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING OGONEK							(U+0328)
		\pgfparserdef{tcparser}{ini}į {\TC@write{ɪ̨}}
		\pgfparserdef{tcparser}{ini}Į {\TC@write{Į}}

	%	LATIN LETTER SMALL CAPITAL I						(U+026A)
	%	with	COMBINING TILDE BELOW						(U+0330)
		\pgfparserdef{tcparser}{ini}ḭ {\TC@write{ɪ̰}}
		\pgfparserdef{tcparser}{ini}Ḭ {\TC@write{Ḭ}}
