
%-----------------------------------------------------------------------------
%
%	LATIN LETTER SMALL CAPITAL O							(U+1D0F)
	\pgfparserdef{tcparser}{ini}o {\TC@write{ᴏ}}
	\pgfparserdef{tcparser}{ini}O {\TC@write{O}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING GRAVE ACCENT						(U+0300)
		\pgfparserdef{tcparser}{ini}ò {\TC@write{ᴏ̀}}
		\pgfparserdef{tcparser}{ini}Ò {\TC@write{Ò}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING ACUTE ACCENT						(U+0301)
		\pgfparserdef{tcparser}{ini}ó {\TC@write{ᴏ́}}
		\pgfparserdef{tcparser}{ini}Ó {\TC@write{Ó}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING CIRCUMFLEX ACCENT					(U+0302)
		\pgfparserdef{tcparser}{ini}ô {\TC@write{ᴏ̂}}
		\pgfparserdef{tcparser}{ini}Ô {\TC@write{Ô}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING TILDE								(U+0303)
		\pgfparserdef{tcparser}{ini}õ {\TC@write{ᴏ̃}}
		\pgfparserdef{tcparser}{ini}Õ {\TC@write{Õ}}

		%	LATIN LETTER SMALL CAPITAL O					(U+1D0F)
		%	with	COMBINING TILDE							(U+0303)
		%	and		COMBINING ACUTE ACCENT					(U+0301)
			\pgfparserdef{tcparser}{ini}ṍ {\TC@write{ᴏ̃́}}
			\pgfparserdef{tcparser}{ini}Ṍ {\TC@write{Ṍ}}

		%	LATIN LETTER SMALL CAPITAL O					(U+1D0F)
		%	with	COMBINING TILDE							(U+0303)
		%	and		COMBINING MACRON						(U+0304)
			\pgfparserdef{tcparser}{ini}ȭ {\TC@write{ᴏ̃̄}}
			\pgfparserdef{tcparser}{ini}Ȭ {\TC@write{Ȭ}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING MACRON							(U+0304)
		\pgfparserdef{tcparser}{ini}ō {\TC@write{ᴏ̄}}
		\pgfparserdef{tcparser}{ini}Ō {\TC@write{Ō}}

		%	LATIN LETTER SMALL CAPITAL O					(U+1D0F)
		%	with	COMBINING MACRON						(U+0304)
		%	and		COMBINING GRAVE ACCENT					(U+0300)
			\pgfparserdef{tcparser}{ini}ṑ {\TC@write{ᴏ̄̀}}
			\pgfparserdef{tcparser}{ini}Ṑ {\TC@write{Ṑ}}

		%	LATIN LETTER SMALL CAPITAL O					(U+1D0F)
		%	with	COMBINING MACRON						(U+0304)
		%	and		COMBINING ACUTE ACCENT					(U+0301)
			\pgfparserdef{tcparser}{ini}ṓ {\TC@write{ᴏ̄́}}
			\pgfparserdef{tcparser}{ini}Ṓ {\TC@write{Ṓ}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING DIAERESIS							(U+0308)
		\pgfparserdef{tcparser}{ini}ö {\TC@write{ᴏ̈}}
		\pgfparserdef{tcparser}{ini}Ö {\TC@write{Ö}}

		%	LATIN LETTER SMALL CAPITAL O					(U+1D0F)
		%	with	COMBINING DIAERESIS						(U+0308)
		%	and		COMBINING MACRON						(U+0304)
			\pgfparserdef{tcparser}{ini}ȫ {\TC@write{ᴏ̈̄}}
			\pgfparserdef{tcparser}{ini}Ȫ {\TC@write{Ȫ}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F
	%	with	COMBINING DOUBLE ACUTE ACCENT				(U+030B)
		\pgfparserdef{tcparser}{ini}ő {\TC@write{ᴏ̋}}
		\pgfparserdef{tcparser}{ini}Ő {\TC@write{Ő}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F
	%	with	COMBINING DOUBLE GRAVE ACCENT				(U+030F)
		\pgfparserdef{tcparser}{ini}ȍ {\TC@write{ᴏ̏}}
		\pgfparserdef{tcparser}{ini}Ȍ {\TC@write{Ȍ}}

	%	LATIN LETTER SMALL CAPITAL O						(U+1D0F)
	%	with	COMBINING SHORT SOLIDUS OVERLAY				(U+0337)
		\pgfparserdef{tcparser}{ini}ø {\TC@write{ᴏ̷}}
		\pgfparserdef{tcparser}{ini}Ø {\TC@write{Ø}}

%-----------------------------------------------------------------------------
